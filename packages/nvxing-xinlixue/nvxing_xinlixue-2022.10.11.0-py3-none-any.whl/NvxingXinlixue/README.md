```
docker run -tid -p <port>:80 apachecn0/nvxing-xinlixue
# 访问 http://localhost:{port} 查看文档
```

```
pip install nvxing-xinlixue
nvxing-xinlixue <port>
# 访问 http://localhost:{port} 查看文档
```
